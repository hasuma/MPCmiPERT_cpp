# Gnuplot script file for plotting data in file "data.txt"
# This file is called plot_sim
set grid
set autoscale
set multiplot layout 3,2
set ylabel 'y(k)'
plot "data.txt" using 1:2 title"{T2-R}" with lines,"data.txt" using 1:3 title"{T4-R}" with lines,"data.txt" using 1:4 title"{T2-M}" with lines,"data.txt" using 1:5 title"{T4-M}" with lines lt 1

set ylabel "pwm(k)"
plot "data.txt" using 1:6 title"PWM-1" with lines,"data.txt" using 1:7 title"PWM-2" with lines lt 4

set ylabel "x(k)"
plot "data.txt" using 1:8 title"x1" with lines,"data.txt" using 1:9 title"x2" with lines,"data.txt" using 1:12 title"x5" with lines,"data.txt" using 1:13 title"x6" with lines lt 2

set ylabel "x(k)"
plot "data.txt" using 1:10 title"x3" with lines,"data.txt" using 1:11 title"x4" with lines lt 5

set ylabel "up(k)"
plot "data.txt" using 1:14 title"up1" with lines,"data.txt" using 1:15 title"up2" with lines, "data.txt" using 1:10 title"x3" with lines,"data.txt" using 1:11 title"x4" with lines lt 3

set ylabel "CPU_TIME(k)"
plot "data.txt" using 1:15 title"CPU_TIME" with lines lt 6

unset multiplot
